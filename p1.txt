// Trace 1
import { chromium } from "playwright";

(async () => {
  const browser = await chromium.launch({ headless: false });
  const page = await browser.newPage();
  try {
    await page.goto("https://login.anz.com/internetbanking");
    await page.getByRole("link", { name: "What’s new" }).click();
    await page.goto("https://www.anz.com.au/support/online-banking/internet-banking/");
    await page.getByRole("link", { name: "Manage your card landing page" }).click();
    await page.goto("https://www.anz.com.au/support/online-banking/manage-your-card/");
    await page.locator("p:nth-of-type(1)").click();
    await page.getByRole("link", { name: "ANZ Falcon®" }).click();
    await page.goto("https://www.anz.com.au/security/how-we-protect-you/falcon/");
    await page.getByRole("link", { name: "Common questions" }).click();
  } finally {
    await browser.close();
  }
})();
